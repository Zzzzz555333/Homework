import tkinter as tk from tkinter import messagebox class NumberGame: def __init__(self, master, num_students): self.master = master self.num_students = num_students self.students = ['未交'] * num_students # 初始化所有学号为“未交” self.buttons = [] self.history = [] # 用来记录操作历史，支持回撤功能 self.create_widgets() def create_widgets(self): self.master.title("学号消消乐") # 创建显示各学号状态的按钮 for i in range(self.num_students): button = tk.Button(self.master, text=f"学号{i+1}\n{self.students[i]}", command=lambda i=i: self.change_status(i), width=20, height=3) button.grid(row=i // 5, column=i % 5) self.buttons.append(button) # 创建回撤按钮 self.undo_button = tk.Button(self.master, text="回撤", command=self.undo, width=20, height=3) self.undo_button.grid(row=(self.num_students // 5) + 1, column=0, columnspan=5) # 创建总结按钮 self.summary_button = tk.Button(self.master, text="生成总结", command=self.generate_summary, width=20, height=3) self.summary_button.grid(row=(self.num_students // 5) + 2, column=0, columnspan=5) def change_status(self, i): """更改学号状态: 未交 -> 全对 -> 需订正""" if self.students[i] == '未交': new_status = '全对' elif self.students[i] == '全对': new_status = '需订正' else: new_status = '未交' # 保存当前状态到历史记录 self.history.append(self.students[:]) # 保存当前状态的快照 self.students[i] = new_status self.update_buttons() def undo(self): """回撤到上一步""" if self.history: self.students = self.history.pop() self.update_buttons() else: messagebox.showinfo("提示", "没有更多的操作历史可以回撤!") def generate_summary(self): """生成总结并显示""" summary = {"未交": 0, "全对": 0, "需订正": 0} for status in self.students: summary[status] += 1 messagebox.showinfo("学号总结", f"未交: {summary['未交']}\n全对: {summary['全对']}\n需订正: {summary['需订正']}") def update_buttons(self): """更新界面上的按钮文本""" for i, button in enumerate(self.buttons): button.config(text=f"学号{i+1}\n{self.students[i]}") if __name__ == "__main__": root = tk.Tk() num_students = int(input("请输入班级人数: ")) game = NumberGame(root, num_students) root.mainloop()
